//Basic Datatypes in C

keyword				Size			Format Specifier
int			atleast 2, usually 4		%d or %i
long int		atleast 4, usually 8		%ld or %li
long long int	     	atleast 8			%lld or %lli

short int		     usually 2			%hd

unsigned int		atleast 2, usually 4		%u
unsigned long int	atleast 4			%lu
unsigned long long int	atleast 8			%llu

float				4			%f
double				8			%lf
long double		atleast 10, usually 12 or 16	%Lf

char				1			%c
signed char			1			%c
unsigned char			1			%c
